Teorie Matrici

Definitie:
Matricea sau tabloul bidimensional, este un tablou cu elemente de acelati tip, dispuse pe linii si coloane.
Fiecare element al matricii se afla pe o anumita linie si pe o anumita coloana.

Declararea:
Matricea se declara folosind comanda: tip nume_matrice[numar_max_linii][nr_max_col];.
Exemplu: int a[2][3] - aceasta este o matrice cu 2 linii si 3 coloane, iar elementele din matrice
vor fi numai de tipul int.

Exemplu:
int a[4][3];
Matricea de mai jos are 4 linii si 3 coloane.

1 3 9
2 5 0
8 9 2
4 6 7

Parcurgerea unei matrici:
Pentru a parcurge o matrice ai nevoie de 2 for-uri, pentru a parcurge de pe fiecare linie, fiecare coloana:

    for(int i = 1;i <= n; i++){ // mergem pe fiecare linie
    
        for(int j = 1;j <= m; j++){ // mergem pe fiecare coloana de pe linia i
        
            // instructiunea pt fiecare caseta din matrice
            // poti da cin >> a[i][j] ca sa-i dai o valoare lui a[i][j];
            // poti da cout << a[i][j]
            // sau orice altceva
            
        }
        
    }
    
    return 0;
}

Diagonalele in matrice:
In primul rand, ca sa vorbim despre diagonalele unei matrici, avem nevoie de o matrice patratica, adica numarul de linii este egal cu numarul de coloane, deoarece numai in matrici patratice putem avea diagonale.

Exemple de matrici patratice:

n = 2
1 7
3 5

n = 5
1 7 2 0 1
8 2 3 9 6
1 8 5 3 0
3 9 3 1 7
4 9 4 1 3

Pentru a parcurge diagonala principala, adica din stanga sus pana jos dreapta, avem nevoie de un singur for, si il vom prelucra pe A[i][i].

Mai jos este un exemplu de cod care citeste o matrice cu 10 linii si coloane si afiseaza elementele de pe diagonala principala. 

#include <iostream>

using namespace std;

int main()
{
  int A[101][101]; // declaram matricea cu maxim 100 linii si coloane

  int n = 10; // matricea va avea 10 linii si 10 coloane

  for (int i = 1; i <= n; i++)
    for (int j = 1; j <= n; j++) cin >> A[i][j]; // citim matricea

  for (int i = 1; i <= n; i++) // parcurgem toata matricea pe diagonala
  {
    cout << A[i][i] << ' '; // afisam elementele de pe diagonala
  }

  return 0;
}

Pentru a parcurge o matrice pe diagonala secundara, adica din stanga jos spre dreapta sus, avem nevoie tot de un singur for, doar ca de data asta mergem de la i = 1 la 1, adica invers, si il vom prelucra pe A[i][n+1-i].


Folosim n + 1 - i deoarece, daca de exemplu vrem sa afisam primul element de pe diagonala secundara a unei matrici cu 5 linii si coloane vom afisa pe A[5][1], de unde 1 = 5 + 1 - 5; daca vrem sa afisam pe al doilea element de pe diagonala vom afisa pe A[4][2], de unde 2 = 5 + 1 - 4, adica n + 1 - i;



Mai jos este un exemplu de cod care citeste o matrice de 10 linii si coloane si va afisa toate elementele de pe diagonala secundara. 

#include <iostream>

using namespace std;

int main()
{
  int A[101][101]; // declaram matricea cu maxim 100 linii si coloane

  int n = 10; // matricea va avea 10 linii si 10 coloane

  for (int i = 1; i <= n; i++)
    for (int j = 1; j <= n; j++) cin >> A[i][j]; // citim matricea

  for (int i = n; i >= 1; i--) // parcurgem toata matricea pe diagonala
  {
    cout << A[i][n+1-i] << ' '; // afisam elementele de pe diagonala secundara
  }

  return 0;
}

Problema #666 NrPrime

#include <iostream>

using namespace std;

// functie pentru a afla daca un nr este prim
// daca prim(numar) == 0 , nu este prim
// daca prim(numar) == 1 , este prim
int prim(int a) {
    if (a < 2) return 0;
    for (int i = 2; i * i <= a; i++)
        if (a % i == 0) return 0;
    return 1;
}

int main() {
    int n; cin >> n;
    int m; cin >> m;

    int nrPrime = 0;
    int A[101][101];

    // citim matricea si facem adunarea deodata
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++) {
            cin >> A[i][j];
            if (i % 2 == 0 and prim(A[i][j]) == 1) nrPrime++;
            // daca este prim si se afla pe o linie cu indice par, nrPrime = nrPrime + 1
        }

    cout << nrPrime;

    return 0;
}

Problema #771 OrdLin

#include<iostream>

using namespace std;

int main() {
    int n; cin >> n;
    int m; cin >> m;

    int a[101][101];
    int b[101];

    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++){
            cin >> a[i][j];
            // facem suma pentru fiecare linie si o tinem in b[i], adica suma elementelor de pe pozitia i
            b[i] += a[i][j];
        }

    for (int i = 1; i < n; i++) // verificam fiecare linie
        for (int j = i + 1; j <= n; j++) // cu fiecare linie
            if (b[i] > b[j]) {
                for (int k = 1; k <= m; k++)
                    // schimbam elementele de pe linia i cu cele de pe linia j
                    swap(a[i][k], a[j][k]);
                swap(b[i], b[j]); // schimbam sumele celor 2 linii intre ele
            }

    // afisarea matricei
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) cout << a[i][j] << " ";
        cout << endl;
    }

    return 0;
}

Problema #769 OrdCol

#include <iostream>

using namespace std;

// functie care schimba toate elementele intre ele de pe 2 coloane( k , j)
void mutare(int a[101][101], int n, int k, int j) {
    for (int i = 1; i <= n; i++)
        swap(a[i][k], a[i][j]);
}

int main() {
    int n; cin >> n;
    int m; cin >> m;

    // citirea matricei
    int A[101][101];
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++) cin >> A[i][j];

    for (int i = 1; i < m; i++)
        for (int j = i + 1; j <= m; j++)
            // daca nu sunt ordonate corect elementele de linia 1 vor schimba coloanele intre ele
            if (A[1][i] > A[1][j]) mutare(A, n, i, j);

    // afisarea matricei
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) cout << A[i][j] << ' ';
        cout << '\n';
    }

    return 0;
}
